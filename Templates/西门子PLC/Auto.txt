// {ProgramName}
// 生成时间：{GeneratedAt}
// DriveDB：{DriveDb}_DriveControl

//清除流程步
IF NOT {ControlDb}.Mode.Status.HomeOK THEN
    Auto[{StationNo}].Step:=0;
    Auto[{StationNo}].Running:=FALSE;
END_IF

//单元周期计时
IF Auto[{StationNo}].Running AND M100.0 THEN
    Auto[{StationNo}].CycleTime:=Auto[{StationNo}].CycleTime+1;
END_IF

//工程启动条件
Start[{StationNo}]:=FALSE;

//工程启动
IF {ControlDb}.Mode.Status.AutoRunning AND {ControlDb}.Mode.Status.HomeOK
    AND Init[{StationNo}].Complete AND NOT Auto[{StationNo}].Running THEN
    IF (Start[{StationNo}] AND NOT {ControlDb}.Mode.Status.StationMode) OR
        (Auto[{StationNo}].Start AND {ControlDb}.Mode.Status.StationMode) THEN
        Auto[{StationNo}].Running:=TRUE;
        Auto[{StationNo}].Step:={FirstStep};
        Auto[{StationNo}].CycleTime:=0;
        Auto[{StationNo}].Comment:="{ProgramName} 启动";
    END_IF
END_IF

RunAgin://再次执行Case条件命令，节省一个扫描周期
Auto[{StationNo}].StepBefor:=Auto[{StationNo}].Step;

// 流程不暂停条件
Auto[{StationNo}].NoPausedRunning:=TRUE;

//工程运转中
IF ({ControlDb}.Mode.Status.AutoRunning AND {ControlDb}.Mode.Status.StepRunning) OR Auto[{StationNo}].NoPausedRunning THEN
    CASE Auto[{StationNo}].Step OF
{Steps}
        1000:
            Auto[{StationNo}].Comment:="自动运行结束";
            Auto[{StationNo}].Running:=FALSE;
            Auto[{StationNo}].Step:=0;
    ELSE
        ;
    END_CASE
END_IF

//CASE命令中步跳转之后，会在下一个扫描周期后再执行下一流程命令
//当满足跳转条件之后，再次执行流程步
IF Auto[{StationNo}].StepBefor<>Auto[{StationNo}].Step THEN
    JMP RunAgin;
END_IF
